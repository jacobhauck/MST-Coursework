% Python style for highlighting
\newcommand\pythonstyle{\lstset{
	language=Python,
	basicstyle=\ttm,
	upquote=true,
	morekeywords={self, assert, True, False, None, as},   % Add keywords here
	keywordstyle=\ttb\color{deepblue},
	commentstyle=\color{orange},
	emph={__init__, BaseException, float, int, str},          % Custom highlighting
	emphstyle=\ttb\color{deepred},    % Custom highlighting style
	stringstyle=\color{deepgreen},          % Any extra options here
	showstringspaces=false,
	numbers=left,
	backgroundcolor=\color{light-gray},
	framexleftmargin=0.5ex,
	framextopmargin=0.5em
}}

\newcommand\txtstyle{\lstset{
	basicstyle=\ttm,
	upquote=true,
	numbers=left,
	breaklines=true,
	postbreak=\mbox{\textcolor{red}{$\hookrightarrow$}\space},
	backgroundcolor=\color{light-gray},
	framexleftmargin=0.5ex,
	framextopmargin=0.5em
}}
